local function startTeleport()
    if teleporting then return end
    teleporting = true

    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()
    local hrp = character:WaitForChild("HumanoidRootPart")

    teleportThread = task.spawn(function()
        local index = 1
        while teleporting do
            local targetPos = coordinates[index]
            local startTime = tick()
            while tick() - startTime < 3 and teleporting do
                hrp.CFrame = CFrame.new(targetPos)
                task.wait(0.1)
            end
            index = index + 1
            if index > #coordinates then
                index = 1
            end
        end
    end)
end

-- H√†m d·ª´ng
local function stopTeleport()
    teleporting = false
    if teleportThread then
        task.cancel(teleportThread)
        teleportThread = nil
    end
end

-- Toggle t·ªïng
local function toggleTeleport()
    if teleporting then
        stopTeleport()
        print("Auto Farm Iron: T·∫ÆT")
    else
        startTeleport()
        print("Auto Farm Iron: B·∫¨T")
    end
end

local function updatePlayerList()
    playerList = {} -- X√≥a danh s√°ch c≈©
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(playerList, player.Name)
        end
    end
    if playerDropdown then
        playerDropdown:SetOptions(playerList) -- C·∫≠p nh·∫≠t dropdown v·ªõi danh s√°ch m·ªõi
    end
end
